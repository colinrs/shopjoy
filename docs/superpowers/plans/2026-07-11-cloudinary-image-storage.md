# Cloudinary Image Storage Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Replace `admin` service's local-only image upload pipeline with a multi-driver (Local + Cloudinary) storage layer where the new Cloudinary driver supports **signed browser-direct upload**, persisted via a new `media_assets` table. Ship a reusable Vue 3 `ImageUploader` component on the shop-admin frontend.

**Architecture:** Backend keeps the existing `storage.Storage` interface but adds `Asset`/`AssetDraft`/`RemoteAsset` and a `Signer` interface. Two drivers (`localStorage`, `cloudinaryStorage`) are wired through `factory.NewStorage`. New endpoints (`POST /uploads/sign`, `POST /uploads/confirm`, `GET /uploads/:id`, plus the existing `POST /uploads` and `DELETE /uploads/:id`) coordinate the sign → direct-upload → confirm flow. Frontend exposes `useImageUpload` composable + `ImageUploader` (single + multi mode) + `ImageCropperDialog`.

**Tech Stack:** Go 1.x, go-zero, GORM, `github.com/cloudinary/cloudinary-go/v2`, MySQL/MariaDB; Vue 3, TypeScript, `vuedraggable`, `cropperjs`, `browser-image-compression`, Element Plus.

**Spec:** `docs/superpowers/specs/2026-07-11-cloudinary-image-storage-design.md`

// admin/internal/infrastructure/persistence/media_asset_repository.go
package persistence

import (
    "context"
    "errors"
    "strings"

    "github.com/go-sql-driver/mysql"
    "gorm.io/gorm"

    "github.com/colinrs/shopjoy/admin/internal/domain/media"
)

type mediaAssetRepo struct {
    db *gorm.DB
}

func NewMediaAssetRepository(db *gorm.DB) media.Repository {
    return &mediaAssetRepo{db: db}
}

func (r *mediaAssetRepo) Insert(ctx context.Context, a *media.Asset) error {
    if err := r.db.WithContext(ctx).Create(a).Error; err != nil {
        var mysqlErr *mysql.MySQLError
        if errors.As(err, &mysqlErr) && mysqlErr.Number == 1062 {
            return errors.New("duplicate media asset")
        }
        return err
    }
    return nil
}

func (r *mediaAssetRepo) FindByID(ctx context.Context, id int64) (*media.Asset, error) {
    var a media.Asset
    if err := r.db.WithContext(ctx).First(&a, id).Error; err != nil {
        if errors.Is(err, gorm.ErrRecordNotFound) {
            return nil, errors.New("asset not found")
        }
        return nil, err
    }
    return &a, nil
}

func (r *mediaAssetRepo) FindByPublicID(ctx context.Context, provider, publicID string) (*media.Asset, error) {
    var a media.Asset
    if err := r.db.WithContext(ctx).
        Where("provider = ? AND public_id = ? AND deleted_at IS NULL", provider, publicID).
        First(&a).Error; err != nil {
        if errors.Is(err, gorm.ErrRecordNotFound) {
            return nil, errors.New("asset not found")
        }
        return nil, err
    }
    return &a, nil
}

func (r *mediaAssetRepo) SoftDelete(ctx context.Context, id int64) error {
    if err := r.db.WithContext(ctx).Delete(&media.Asset{}, id).Error; err != nil {
        if strings.Contains(err.Error(), "not found") {
            return errors.New("asset not found")
        }
        return err
    }
    return nil
}
```

// admin/internal/infrastructure/storage/storage.go
package storage

import (
    "context"
    "io"
    "mime/multipart"
    "time"
)

// Category 图片分类。
type Category string

const (
    CategoryProduct Category = "product"
    CategoryBanner  Category = "banner"
    CategoryAvatar  Category = "avatar"
)

// Asset backend-managed asset metadata.
type Asset struct {
    ID        string
    PublicID  string
    URL       string
    Filename  string
    Size      int64
    MimeType  string
    Width     int
    Height    int
    Format    string
    Category  Category
    Provider  string // "local" | "cloudinary"
    TenantID  int64
    CreatedBy int64
    CreatedAt time.Time
}

// AssetDraft Save() input — caller provides reader.
type AssetDraft struct {
    Filename  string
    Reader    io.Reader
    MimeType  string
    Category  Category
    TenantID  int64
    CreatedBy int64
}

// RemoteAsset frontend finished direct upload and reports back.
type RemoteAsset struct {
    PublicID  string
    URL       string
    Filename  string
    Size      int64
    MimeType  string
    Width     int
    Height    int
    Format    string
    Category  Category
    TenantID  int64
    CreatedBy int64
}

// Storage 资源存储接口。
type Storage interface {
    Save(ctx context.Context, draft AssetDraft) (*Asset, error)
    RegisterAsset(ctx context.Context, remote RemoteAsset) (*Asset, error)
    Delete(ctx context.Context, id string) error
    Get(ctx context.Context, id string) (*Asset, error)
    GetURL(ctx context.Context, id string) (string, error)
}

// Signer 远端签名能力（仅 cloudinaryStorage 实现）。
type Signer interface {
    Sign(ctx context.Context, p SignParams) (Signature, error)
}

// SignParams is the requested signature context.
type SignParams struct {
    Category  Category
    TenantID  int64
    Filename  string
    Timestamp int64
}

// Signature is the public-safe bundle returned to the browser.
type Signature struct {
    CloudName    string `json:"cloud_name"`
    APIKey       string `json:"api_key"`
    Timestamp    string `json:"timestamp"`
    Signature    string `json:"signature"`
    Folder       string `json:"folder"`
    PublicID     string `json:"public_id"`
    UploadPreset string `json:"upload_preset,omitempty"`
}

// FileHeaderCompat is a small adapter so caller-side logic keeps working
// with multipart headers without forcing storage to depend on multipart package elsewhere.
//
//   The existing UploadLogic reads the file and detects MIME; it then opens
//   the file and passes through to Storage.Save with a reader. The compat
//   helper below turns a *multipart.FileHeader into an AssetDraft quickly.
func FromMultipart(file *multipart.FileHeader, category Category, tenantID, createdBy int64) (AssetDraft, io.ReadCloser, error) {
    src, err := file.Open()
    if err != nil {
        return AssetDraft{}, nil, err
    }
    return AssetDraft{
        Filename:  file.Filename,
        Reader:    src,
        MimeType:  file.Header.Get("Content-Type"),
        Category:  category,
        TenantID:  tenantID,
        CreatedBy: createdBy,
    }, src, nil
}
```

```go
// admin/internal/infrastructure/storage/local.go
package storage

import (
    "context"
    "errors"
    "fmt"
    "image"
    "io"
    "os"
    "path/filepath"
    "strings"
    "time"

    "github.com/google/uuid"

    "github.com/colinrs/shopjoy/admin/internal/domain/media"
    snowflake "github.com/colinrs/shopjoy/pkg/snowflake"
)

type localStorage struct {
    basePath string
    repo     media.Repository
    idGen    snowflake.Snowflake
}

func newLocal(cfg Config, repo media.Repository, idGen snowflake.Snowflake) *localStorage {
    base := cfg.Local.BasePath
    if base == "" {
        base = "./uploads"
    }
    return &localStorage{basePath: base, repo: repo, idGen: idGen}
}

func (s *localStorage) Save(ctx context.Context, draft AssetDraft) (*Asset, error) {
    if draft.Reader == nil {
        return nil, errors.New("reader is nil")
    }
    ext := strings.ToLower(filepath.Ext(draft.Filename))
    if ext == "" || !strings.Contains(".jpg.jpeg.png.gif.webp", ext) {
        return nil, errors.New("unsupported extension")
    }

    now := time.Now().UTC()
    dir := filepath.Join(s.basePath, string(draft.Category),
        fmt.Sprintf("%d", now.Year()),
        fmt.Sprintf("%02d", int(now.Month())),
        fmt.Sprintf("%02d", now.Day()),
    )
    if err := os.MkdirAll(dir, 0o750); err != nil {
        return nil, fmt.Errorf("mkdir: %w", err)
    }
    assetID := s.idGen.Next()
    assetIDStr := fmt.Sprintf("img_%d", assetID)
    filePath := filepath.Join(dir, assetIDStr+ext)

    absBase, _ := filepath.Abs(s.basePath)
    absFile, _ := filepath.Abs(filePath)
    if !strings.HasPrefix(absFile, absBase) {
        return nil, errors.New("path traversal blocked")
    }

    dst, err := os.Create(filePath) //nolint:gosec
    if err != nil {
        return nil, fmt.Errorf("create file: %w", err)
    }
    defer dst.Close()

    width, height := 0, 0
    if ext != ".gif" && ext != ".webp" {
        if cfg, _, derr := image.DecodeConfig(draft.Reader); derr == nil {
            width, height = cfg.Width, cfg.Height
        }
        if _, serr := draft.Reader.(io.Seeker); serr == nil {
            // not a seeker; copy writer first
        }
    }
    if _, err := io.Copy(dst, draft.Reader); err != nil {
        return nil, fmt.Errorf("write file: %w", err)
    }

// Sign produces a Cloudinary-compatible signed bundle for browser direct upload.
func (s *cloudinaryStorage) Sign(ctx context.Context, p SignParams) (Signature, error) {
    if s.cfg.APISecret == "" {
        return Signature{}, errors.New("cloudinary api_secret missing")
    }
    ts := p.Timestamp
    if ts == 0 {

    params := url.Values{}
    params.Set("folder", folder)

    return Signature{
        CloudName:    s.cfg.CloudName,

// canonicalParams joins a sorted url.Values as "k=v&k=v".
func canonicalParams(v url.Values) string {
    keys := make([]string, 0, len(v))
    for k := range v {
        keys = append(keys, k)
    }
    sort.Strings(keys)
    parts := make([]string, 0, len(keys))
    for _, k := range keys {
        parts = append(parts, k+"="+v.Get(k))
    }
    return strings.Join(parts, "&")
}

func sha1Hex(s string) string {
    h := sha1.Sum([]byte(s))
    return hex.EncodeToString(h[:])
}

import { nanoid } from 'nanoid'
import { signImage, confirmImage, deleteImage } from '@/api/upload'
import imageCompression from 'browser-image-compression'
import axios from 'axios'

export type UploadStatus =
  | 'pending'
  | 'signing'
  | 'uploading'
  | 'confirming'
  | 'done'
  | 'error'
  | 'cancelled'

export interface UploadItem {
  uid: string
  raw: File
  preview: string
  status: UploadStatus
  progress: number
  url?: string
  assetId?: string
  publicId?: string
  error?: string
  controller?: AbortController
}

export interface UseImageUploadOptions {
  category: string
  maxSize?: number
  concurrency?: number
  compress?: boolean
}

export function useImageUpload(opts: UseImageUploadOptions) {
  const items: Ref<UploadItem[]> = ref([])
  const maxSize = opts.maxSize ?? 5 * 1024 * 1024
  const concurrency = opts.concurrency ?? 3
  const compress = opts.compress ?? true

  let activeCount = 0
  const queue: UploadItem[] = []

  const next = () => {
    while (activeCount < concurrency && queue.length) {
      const it = queue.shift()!
      activeCount++
      void runOne(it).finally(() => {
        activeCount--
        next()
      })
    }
  }

  async function runOne(it: UploadItem) {
    const ctrl = new AbortController()
    it.controller = ctrl
    try {
      it.status = 'signing'
      const sig = await signImage({
        category: opts.category,
        filename: it.raw.name,
        mime_type: it.raw.type,
      })

      if (ctrl.signal.aborted) { it.status = 'cancelled'; return }

      it.status = 'uploading'
      const fd = new FormData()
      fd.append('file', it.raw)
      fd.append('api_key', sig.api_key)
      fd.append('timestamp', sig.timestamp)
      fd.append('signature', sig.signature)
      fd.append('folder', sig.folder)
      fd.append('public_id', sig.public_id)
      if (sig.upload_preset) fd.append('upload_preset', sig.upload_preset)

      const res = await axios.post(sig.upload_url, fd, {
        signal: ctrl.signal,
        onUploadProgress: (e) => {
          if (e.total) it.progress = Math.round((e.loaded / e.total) * 100)
        },
      })
      const cloud = res.data as {
        public_id: string; secure_url: string; bytes: number;
        width: number; height: number; format: string;
      }

      if (ctrl.signal.aborted) { it.status = 'cancelled'; return }
      it.status = 'confirming'
      const out = await confirmImage({
        asset_id: sig.asset_id,
        public_id: cloud.public_id,
        url: cloud.secure_url,
        filename: it.raw.name,
        size: cloud.bytes,
        mime_type: it.raw.type,
        width: cloud.width,
        height: cloud.height,
        format: cloud.format,
        category: opts.category,
      })

      it.url = out.url
      it.assetId = out.id
      it.publicId = cloud.public_id
      it.progress = 100
      it.status = 'done'
    } catch (err: any) {
      if (err?.name === 'CanceledError' || err?.name === 'AbortError') {
        it.status = 'cancelled'
      } else {
        it.status = 'error'
        it.error = err?.message ?? 'upload failed'
      }
    }
  }

  async function uploadOne(file: File): Promise<UploadItem> {
    let prepared = file
    if (compress && file.size > 1024 * 1024 && /^image\//.test(file.type)) {
      try {
        prepared = await imageCompression(file, { maxSizeMB: 1, maxWidthOrHeight: 2400, useWebWorker: true })
      } catch { /* ignore compress failure */ }
    }
    if (prepared.size > maxSize) {
      const it: UploadItem = {
        uid: nanoid(), raw: prepared, preview: URL.createObjectURL(prepared),
        status: 'error', progress: 0, error: 'exceeds max size',
      }
      items.value.push(it)
      return it
    }
    const it: UploadItem = {
      uid: nanoid(), raw: prepared, preview: URL.createObjectURL(prepared),
      status: 'pending', progress: 0,
    }
    items.value.push(it)
    queue.push(it)
    next()
    return it
  }

  async function remove(it: UploadItem): Promise<void> {
    it.controller?.abort()
    if (it.assetId) await deleteImage(it.assetId).catch(() => undefined)
    URL.revokeObjectURL(it.preview)
    items.value = items.value.filter(x => x.uid !== it.uid)
  }

  function clearAll() {
    for (const it of items.value) {
      it.controller?.abort()
      URL.revokeObjectURL(it.preview)
    }
    items.value = []
    queue.length = 0
  }

  return { items, uploadOne, remove, clearAll }
}
```

      v-if="multiple && (!max || itemList.length < max)"
      class="image-uploader__dropzone"
      :class="{ 'is-dragover': dragover }"

      <i class="el-icon-upload" />
      <p>点击或拖拽图片到此处</p>
      <p class="hint">支持 {{ accept }}，单文件不超过 {{ formatBytes(maxSize) }}</p>
    </div>

    <image-cropper-dialog

      :visible.sync="cropperVisible"
      :src="cropTarget.preview"
      :title="cropTarget.raw.name"
      :ratio="cropRatio"
      @confirm="onCropConfirm"
      @cancel="cropTarget = null"

  </div>

<script setup lang="ts">
import { computed, ref, watch } from 'vue'
import draggable from 'vuedraggable'
import { ElMessage } from 'element-plus'
import ImageCropperDialog from './ImageCropperDialog.vue'
import { useImageUpload, type UploadItem } from '@/composables/useImageUpload'

const props = defineProps<{
  modelValue: string | string[]
  category: string
  multiple?: boolean
  max?: number
  maxSize?: number
  accept?: string
  compress?: boolean
  crop?: boolean
  cropRatio?: number
}>()
const emit = defineEmits<{ 'update:modelValue': [string | string[]] }>()

const accept = props.accept ?? 'image/jpeg,image/png,image/webp,image/gif'
const maxSize = props.maxSize ?? 5 * 1024 * 1024
const compress = props.compress ?? true
const crop = props.crop ?? false
const cropRatio = props.cropRatio ?? 1

const { items, uploadOne, remove } = useImageUpload({
  category: props.category, maxSize, compress,
})

const fileInput = ref<HTMLInputElement | null>(null)
const dragover = ref(false)
const cropTarget = ref<UploadItem | null>(null)
const cropperVisible = ref(false)

// modelValue ⇆ items: 把字符串 URL 列表跟 items 里已经上传的 url 关联
// 简化策略：每次 modelValue 变化，重置 items（不重新上传已存在的 url）。
watch(() => props.modelValue, (v) => {
  const urls = Array.isArray(v) ? v : v ? [v] : []
  if (urls.length === 0 && items.value.length > 0) return
  // 外部传入的 url 视为已 done —— 不重置未完成 item
}, { immediate: true })

const itemList = computed({
  get: () => items.value,
  set: (v) => { items.value = v; emitChange() },
})

function emitChange() {
  const urls = items.value.filter(i => i.status === 'done' && i.url).map(i => i.url!)
  if (props.multiple) emit('update:modelValue', urls)
  else emit('update:modelValue', urls[0] ?? '')
}

function pick() { (fileInput.value as HTMLInputElement)?.click() }
async function onPick(e: Event) {
  const input = e.target as HTMLInputElement
  if (!input.files) return
  for (const f of Array.from(input.files)) await addFile(f)
  input.value = ''
  emitChange()
}
async function onDrop(e: DragEvent) {
  dragover.value = false
  if (!e.dataTransfer) return
  for (const f of Array.from(e.dataTransfer.files)) await addFile(f)
  emitChange()
}
async function addFile(file: File) {
  if (props.max && items.value.length >= props.max) {
    ElMessage.warning(`最多上传 ${props.max} 张图片`)
    return
  }
  await uploadOne(file)
}
function statusLabel(s: string) {
  return { pending: '等待', signing: '签名', uploading: '上传中', confirming: '确认', done: '完成', error: '失败', cancelled: '取消' }[s] ?? s
}
async function removeAt(it: UploadItem) {
  await remove(it)
  emitChange()
}
async function retry(it: UploadItem) {
  await uploadOne(it.raw)
  emitChange()
}
function openCrop(it: UploadItem) {
  cropTarget.value = it
  cropperVisible.value = true
}
async function onCropConfirm(blob: Blob) {
  if (!cropTarget.value) return
  const file = new File([blob], cropTarget.value.raw.name, { type: blob.type })
  await remove(cropTarget.value)
  await uploadOne(file)
  cropTarget.value = null
  emitChange()
}
function formatBytes(n: number) {
  if (n >= 1024 * 1024) return `${(n / 1024 / 1024).toFixed(1)} MB`
  return `${(n / 1024).toFixed(0)} KB`
}
</script>
